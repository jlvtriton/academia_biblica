# ⚙️ JAVASCRIPT DEL QUIZ

```html
<script>

const quizButtons = document.querySelectorAll('.quiz-btn');

quizButtons.forEach(button => {

  button.addEventListener('click', () => {

    const quizBox = button.closest('.quiz-box');
    const result = quizBox.querySelector('.quiz-result');
    const buttons = quizBox.querySelectorAll('.quiz-btn');

    if (quizBox.classList.contains('answered')) {
      return;
    }

    quizBox.classList.add('answered');

    buttons.forEach(btn => {
      btn.disabled = true;
    });

    if(button.classList.contains('correct')){

      result.innerHTML = '✅ ¡Correcto!';
      result.style.color = '#22c55e';

      button.classList.add('btn-success');

    } else {

      result.innerHTML = '❌ Respuesta incorrecta';
      result.style.color = '#ef4444';

      button.classList.add('btn-danger');

      buttons.forEach(btn => {
        if(btn.classList.contains('correct')){
          btn.classList.add('btn-success');
        }
      });

    }

  });

});

</script>
```
